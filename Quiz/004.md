number = input("Please enter an integer ")
max_num_tries = 5
while not number.isdigit() and max_num_tries>0:
    max_num_tries = max_num_tries - 1
    number = input(f"Error you have {max_num_tries} chances left. Please enter an integer ")
if max_num_tries !=0:
    print("Good job")

    exit()
