def doors(number_doors:int):
    doors = []
    for student in range(number_doors):
        doors.append(False)
    for door in range(1,(number_doors+1)):
        for x in range(1,number_doors):
            if x % door == 0:
                doors[x-1] = not doors[x-1]
                num_turns = doors.count(True)
    return(num_turns)
