word = input()
for i in range(len(word)-1):
    letter = word[i]
    print(letter.upper())

sum = 0

for l in word:
    code = ord(l)
    if code<91:
        sum = sum + (code - 90)
    if code>96:
        sum = sum + (code-96)
print(f"The code for {word} is {sum}")
